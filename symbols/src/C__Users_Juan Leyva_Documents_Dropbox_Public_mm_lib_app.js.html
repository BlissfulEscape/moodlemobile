 19 /**
 20  * @fileOverview This is the js file that loads all the dependencies and init the application.
 21  * @author <a href="mailto:jleyva@cvaconsulting.com">Juan Leyva</a>
 22  * @version 1.2
 23  */
 24 
 25 // Base path for all the files required.
 26 require.config({
 27     paths: {
 28         root: '..'
 29     }
 30 });
 31 
 32 // Requeriments for launch the app, the function is not executed until both files are fully loaded.
 33 // We need at least the config.json file with all the settings and the language file.
 34 require(['root/externallib/text!root/config.json', 'root/externallib/text!root/lang/en.json'],
 35 function(config, lang) {
 36 
 37     config = JSON.parse(config);
 38     // Init the app.
 39     MM.init(config);
 40     MM.lang.base = JSON.parse(lang);
 41 
 42     // Once the config and base lang are loaded, we load all the plugins defined in the config.json file.
 43     require.config({
 44         baseUrl: 'plugins',
 45         packages: config.plugins
 46     });
 47 
 48     // We load extra languages if are present in the config file.
 49     var extraLang = 'root/externallib/text!root/lang/' + config.default_lang + '.json';
 50     config.plugins.unshift(extraLang);
 51 
 52     require(config.plugins,
 53         function(extraLang) {
 54             MM.lang.loadLang('core', config.default_lang, JSON.parse(extraLang));
 55             $(document).ready(function() {
 56                 // Load the base layout of the app.
 57                 MM.loadLayout();
 58 
 59                 // Register the device in the notifications server, retrieve tokens
 60                 MM.plugins.notifications.registerDevice();
 61 
 62                 // Check for notification when the app launch.
 63                 MM.plugins.notifications.check();
 64 
 65                 // Listen for push events.
 66                 MM.plugins.notifications.listenEvents();
 67             });
 68         }
 69     );
 70 });
