 25 /**
 26   * @namespace Holds all the MoodleMobile cache functionallity.
 27  */
 28 MM.cache = {
 29 
 30     /**
 31      * Retrieves an element from the cache
 32      *
 33      * @param {string} id The unique id for the cached element.
 34      * @param {boolean} omitExpires If true, we retrieve the element even if the cache is expired.
 35      */
 36     getElement: function(id, omitExpires) {
 37 
 38         MM.log('Cache: Trying to get a cached element, id: ' + id);
 39 
 40         if (typeof omitExpires == 'undefined') {
 41             omitExpires = false;
 42         }
 43 
 44         // Cache elements has a prefix indicating the current site.
 45         id = MM.config.current_site.id + '-' + id;
 46         cache = MM.db.get('cache', id);
 47 
 48         if (typeof cache == 'undefined' || !cache) {
 49             MM.log('Cache: Cached element not found, id: ' + id);
 50             return false;
 51         }
 52 
 53         cache = cache.toJSON();
 54 
 55         var d = new Date();
 56         var now = d.getTime();
 57 
 58         if (!omitExpires) {
 59             if (now > cache.mmcacheexpirationtime) {
 60                 return false;
 61             }
 62         }
 63 
 64         if (typeof cache.data != 'undefined') {
 65             var expires = (cache.mmcacheexpirationtime - now) / 1000;
 66             MM.log('Cache: Cached element found, id: ' + id + ' expires in ' + expires + ' seconds');
 67             return cache.data;
 68         }
 69         return false;
 70     },
 71 
 72     /**
 73      * Add an element to the cache
 74      *
 75      * @param {string} id The unique id for the element.
 76      * @param {object} el The element to be added to the cache.
 77      * @param {string} type The type of element, general, ws, etc..
 78      */
 79     addElement: function(id, el, type) {
 80         MM.log('Cache: Adding element to cache: ' + id);
 81 
 82         if (typeof(type) == 'undefined') {
 83             type = 'general';
 84         }
 85 
 86         id = MM.config.current_site.id + '-' + id;
 87 
 88         var cachedEl = {
 89             id: id,
 90             data: el,
 91             type: type,
 92             mmcacheexpirationtime: 0
 93         };
 94 
 95         var d = new Date();
 96         cachedEl.mmcacheexpirationtime = d.getTime() + MM.getConfig('cache_expiration_time', 0);
 97 
 98         MM.db.insert('cache', cachedEl);
 99         return true;
100     },
101 
102     /**
103      * Add a WS call to the cache
104      *
105      * @param {string} url WS url.
106      * @param {object} data WS parameters,.
107      * @param {object} res Result of the WS call.
108      */
109     addWSCall: function(url, data, res) {
110         MM.log('Cache: Adding a WS cached call: ' + data.wsfunction);
111 
112         var key = hex_md5(url + JSON.stringify(data));
113         MM.cache.addElement(key, res, 'ws');
114         return true;
115     },
116 
117     /**
118      * Retrieves a WS call from the cache
119      *
120      * @param {string} url WS url.
121      * @param {object} data WS parameters.
122      * @param {boolean} omitExpires If true, we retrieve the element even if the cache is expired.
123      */
124     getWSCall: function(url, data, omitExpires) {
125 
126         MM.log('Cache: Trying to get a WS cached call: ' + data.wsfunction);
127 
128         // Unique hash.
129         var key = hex_md5(url + JSON.stringify(data));
130         var el = MM.cache.getElement(key, omitExpires);
131         return el;
132     },
133 
134     /**
135      * Purge all WS cache
136      */
137     purge: function() {
138         MM.log('Cache: purge all caches');
139         MM.db.each('cache', function(el) {
140             if (el.get('type') == 'ws') {
141                 MM.db.remove('cache', el.get('id'));
142             }
143         });
144         MM.popMessage(MM.lang.s('allcachespurged'));
145     }
146 };
