 25 /**
 26   * @namespace Holds all the MoodleMobile db functionallity.
 27  */
 28 MM.db = {
 29     get: function(collection, id) {
 30     	MM.log('DB: Get element from DB ' + collection + ' id> ' + id);
 31         // Fetch all the elements from the collection.
 32         MM.collections[collection].fetch();
 33         return MM.collections[collection].get(id);
 34     },
 35     where: function(collection, conditions) {
 36     	MM.log('DB: Get element from DB using conditiongs');
 37         // Fetch all the elements from the collection.
 38         MM.collections[collection].fetch();
 39         return MM.collections[collection].where(conditions);
 40     },
 41     each: function(collection, fn) {
 42     	MM.log('DB: Get all elements from DB');
 43         // Fetch all the elements from the collection.
 44         MM.collections[collection].fetch();
 45         MM.collections[collection].each(fn);
 46     },
 47     insert: function(collection, model) {
 48     	MM.log('DB: Insert element into DB collection: ' + collection);
 49         return MM.collections[collection].create(model);
 50     },
 51     remove: function(collection, modelId) {
 52     	MM.log('DB: Remove element from DB collection: ' + collection + ' id: ' + modelId);
 53         var model = MM.db.get(collection, modelId);
 54         return model.destroy();
 55     },
 56     length: function(collection) {
 57         MM.collections[collection].fetch();
 58         return MM.collections[collection].length; }
 59 };
