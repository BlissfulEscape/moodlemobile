 25 /**
 26   * @namespace Holds all the MoodleMobile file system functionallity.
 27  */
 28 MM.fs = {
 29 
 30     basePath: '',
 31 
 32 	init: function(callBack) {
 33 		MM.log('FS: Loading File System');
 34 
 35         if (typeof(MM.fs.fileSystemRoot) != 'undefined') {
 36             // The file system is yet loaded.
 37             MM.log('FS: The file system was previously loaded');
 38             callBack();
 39             return;
 40         }
 41 
 42         if (MM.deviceOS == 'android') {
 43             MM.fs.basePath = 'Android/data/' + MM.config.app_id;
 44         }
 45         MM.fs.loadFS(callBack);
 46 	},
 47 
 48     getRoot: function() {
 49         if (!MM.fs.fileSystemRoot) {
 50             MM.fs.loadFS(function() { MM.fs.fileSystemRoot.fullPath });
 51         }
 52         else {
 53             return MM.fs.fileSystemRoot.fullPath;
 54         }
 55     },
 56 
 57     loadFS: function(callBack) {
 58         window.requestFileSystem(LocalFileSystem.PERSISTENT, 0,
 59             function(fileSystem) {
 60                MM.fs.fileSystemRoot = fileSystem.root;
 61                if (MM.deviceOS == 'android') {
 62                 MM.fs.fileSystemRoot.getDirectory(MM.fs.basePath, {create: true, exclusive: false},
 63                     function(entry) {
 64                         MM.fs.fileSystemRoot = entry;
 65                         callBack();
 66                     },
 67                     function() {
 68                         MM.popErrorMessage('Critial error accessing file system, Android/data/app_id does not exists or can be created');
 69                     }
 70                 );
 71                } else {
 72                 callBack();
 73                }
 74            }, function() {
 75               MM.popErrorMessage('Critial error accessing file system');
 76            });
 77     },
 78 
 79     fileExists: function(path, successCallback, errorCallback) {
 80         var directory = path.substring(0, filename.lastIndexOf('/') - 1);
 81         var filename = path.substr(filename.lastIndexOf('/') + 1);
 82         MM.fs.fileSystemRoot.getDirectory(directory, { create: false },
 83             function(entry) {
 84                 entry.getFile(filename, { create: false },
 85                 function(entryFile) {
 86                     successCallback(entryFile.fullPath);
 87                 }
 88                 , errorCallback);
 89             }
 90             , errorCallback);
 91     },
 92 
 93     createDir: function(path, successCallback, dirEntry) {
 94 
 95         path = path.replace('file:///', '');
 96         MM.log('FS: Creating full directory ' + path);
 97 
 98         var paths = path.split('/');
 99 
100         var baseRoot = MM.fs.fileSystemRoot;
101         if (dirEntry) {
102             baseRoot = dirEntry;
103         }
104 
105         MM.log('FS: Creating directory ' + paths[0] + 'in' + baseRoot.fullPath);
106         baseRoot.getDirectory(paths[0], {create: true, exclusive: false},
107             function(newDirEntry) {
108                 if (paths.length == 1) {
109                     successCallback(newDirEntry);
110                     return;
111                 }
112                 // Recursively, create next directories
113                 paths.shift();
114                 MM.fs.createDir(paths.join('/'), successCallback, newDirEntry);
115             },
116             function() {
117                 MM.popErrorMessage('Critial error accessing file system');
118              });
119     }
120 };
