 38     /**
 39      * Initial setup of the app: device type detection, routes, models, settings.
 40      * This function is executed once the config.json file is loaded and previously to loading the app plugins.
 41      *
 42      * @this {MM}
 43      * @param {Object.<>} Settings loaded from /config.json.
 44      */
 45     init: function(config) {
 46         // Config.js settings.
 47         this.config = config;
 48 
 49         MM.log('MM: Initializating app');
 50 
 51         // Device type detection.
 52         if (matchMedia('only screen and (min-width: 768px) and (-webkit-min-device-pixel-ratio: 1)').matches) {
 53             this.deviceType = 'tablet';
 54             $('body').addClass('tablet');
 55         } else {
 56             this.deviceType = 'phone';
 57             $('body').addClass('phone');
 58         }
 59 
 60         //OS Detecting
 61         this.deviceOS = (navigator.userAgent.match(/iPad/i)) == 'iPad' ? 'ios' : (navigator.userAgent.match(/iPhone/i)) == 'iPhone' ? 'ios' : (navigator.userAgent.match(/Android/i)) == 'Android' ? 'android' : 'null';
 62 
 63 
 64         MM.inComputer = navigator.userAgent.indexOf('Chrome') >= 0 ||
 65                         navigator.userAgent.indexOf('Safari') >= 0 ||
 66                         navigator.userAgent.indexOf('MSIE') >= 0 ||
 67                         navigator.userAgent.indexOf('Firefox') >= 0;
 68         MM.inComputer = MM.inComputer && navigator.userAgent.indexOf('Mobile') == -1;
 69 
 70         // If we are testing in a computer, we load the Cordova emulating javascript.
 71         if (MM.inComputer) {

 73             MM.cordova.loadEmulator();
 74         }
 75 
 76         // Load the Backbone.Router for URL routing.
 77         var appRouter = Backbone.Router.extend();
 78         this.Router = new appRouter;
 79 
 80         // AJAX error handling.
 81         $.ajaxSetup({'error': function(xhr,textStatus, errorThrown) {
 82             var error = MM.lang.s('cannotconnect');
 83             if (xhr.status == 404) {
 84                 error = MM.lang.s('invalidscheme');
 85             }
 86             MM.popErrorMessage(error);
 87         }});

 90         // Elements for the core storage model.

 92             setting: {type: 'model', bbproperties: {initialize: function() { MM.config[this.get('name')] = this.get('value'); }}},

116     /**
117      * Checks if the device is connected to Internet
118      * We use Cordova API for that, if the config setting "dev_offline" is set we return allways not connected.
119      *
120      * @return {boolean} True if the device is connected.
121      */
122     deviceConnected: function() {
123         var connected = true;
124 
125         if (typeof(navigator.network) != 'undefined') {
126             var networkState = navigator.network.connection.type;
127             connected = (networkState != Connection.NONE && networkState != Connection.UNKNOWN);
128             MM.log('Internet connection checked ' + connected);
129         }
130 
131         var offline = MM.getConfig('dev_offline');
132         if (typeof(offline) != 'undefined' && offline) {
133             connected = false;
134         }
135 
136         return connected;

403     /**
404      * Saves a site in the database
405      *
406      * @param {string} username User name.
407      * @param {string} password Password.
408      * @param {string} siteurl The site url.
409      * @return {boolean} Allways returns false
410      */
411     saveSite: function(username, password, siteurl) {
412 
413         var loginURL = siteurl + '/login/token.php';
414 
415         // Now, we try to get a valid token.
416         $.getJSON(loginURL,
417             {
418                 username: username,
419                 password: password,
420                 service: MM.config.wsservice
421             }
422             ,function(json) {
423                 if (typeof(json.token) != 'undefined') {
424                     var mytoken = json.token;
425 
426                     MM.setConfig('current_token', mytoken);
427 
428                     var preSets = {
429                         wstoken: mytoken,
430                         siteurl: siteurl,
431                         cache: 0
432                     };
433 
434                     // We have a valid token, try to get the site info.
435                     MM.moodleWSCall('moodle_webservice_get_siteinfo', {}, function(d) {
436                         d.id = hex_md5(d.siteurl + username);
437                         d.token = mytoken;
438                         var site = MM.db.insert('sites', d);
439                         MM.setConfig('current_site', d);
440 
441                         MM.plugins.notifications.registerForPushNotification();
442 
443                         MM.loadSite(site.id);
444 
445                     }, preSets);
446                 }
447                 else {
448                     var error = MM.lang.s('invalidaccount');
449                     if (typeof(json.error) != 'undefined') {
450                         error = json.error;
451                     }
452                     MM.popErrorMessage(error);
453                 }
454             });
455         return false;
456     },

458     /**
459      * Register a plugin in the main Namespaces "MM"
460      * This makes possible to access all the plugin functions and variables from the main namespace.
461      * This function is called when a module is being loaded.
462      *
463      * @param {!Object} And object representing a plugin.
464      */
465     registerPlugin: function(plugin) {
466         var pluginName = plugin.settings.name;
467 
468         if (! plugin.icon) {
469             plugin.settings.icon = 'plugins/' + pluginName + '/icon.png';
470         }
471 
472         // Load the plugin in the main Namespace.
473         this.plugins[pluginName] = plugin;
474 
475         for (var el in plugin.routes) {
476             var route = plugin.routes[el];
477             // Route[0] URL to match, Route[1] id, Route[2] function to call on match.
478             this.Router.route(route[0], route[1], this.plugins[pluginName][route[2]]);
479         }
480         this.loadModels(plugin.storage);
481 
482         // Load default strings.
483         if (plugin.settings.lang.component != 'core') {
484             MM.lang.loadPluginLang(pluginName, JSON.parse(plugin.settings.lang.strings));
485         }
486 
487         // Sync hooks (like cron jobs)
488         if (typeof(plugin.sync) != 'undefined') {
489             MM.sync.registerHook(pluginName, plugin.sync);
490         }
491     },

493     /**
494      * Loads backbone Models
495      *
496      * @param {Array.<Object>} elements The models to be loades.
497      */
498     loadModels: function(elements) {
499 
500         for (var el in elements) {
501             var obj = elements[el];
502 
503             // This allow plugins to load Backbone properties to models and collections.
504             if (typeof obj.bbproperties == 'undefined') {
505                 obj.bbproperties = {};
506             }
507 
508             if (obj.type == 'model') {
509                 this.models[el] = Backbone.Model.extend(obj.bbproperties);
510             }
511             else if (obj.type == 'collection') {
512                 obj.bbproperties.model = this.models[obj.model];
513                 obj.bbproperties.localStorage = new Store(el);
514                 var col = Backbone.Collection.extend(obj.bbproperties);
515                 // Now the instance.
516                 this.collections[el] = new col();
517             }
518         }
519     },

542     /**
543      * A wrapper function for a moodle WebService call.
544      *
545      * @param {string} method The WebService method to be called.
546      * @param {Object} data Arguments to pass to the method.
547      * @param {Object} callBack Function to be called in success.
548      * @param {Object} preSets Extra settings
549      *      cache For avoid using caching
550      *      sync For indicate that is a call in a sync process
551      *      silently For not raising errors.
552      */
553     moodleWSCall: function(method, data, callBack, preSets) {
554 
555         // Force data elements to be string.
556         for (var el in data) {
557             data[el] = data[el] + '';
558         }
559 
560         if (typeof(preSets) == 'undefined') {
561             preSets = {};
562         }
563         if (typeof(preSets.cache) == 'undefined') {
564             preSets.cache = 1;
565         }
566         if (typeof(preSets.sync) == 'undefined') {
567             preSets.sync = 0;
568         }
569         if (typeof(preSets.silently) == 'undefined') {
570             preSets.silently = false;
571         }
572 
573         if (typeof(preSets.wstoken) == 'undefined') {
574             var mytoken = MM.config.current_token;
575             if (!mytoken) {
576                 MM.popErrorMessage('Unexpected error. Please close and open again the application (mytoken)');
577                 return false;
578             }
579         }
580         else {
581             var mytoken = preSets.wstoken;
582         }
583 
584         if (typeof(preSets.siteurl) == 'undefined') {
585             var siteurl = MM.config.current_site.siteurl;
586             if (!siteurl) {
587                 MM.popErrorMessage('Unexpected error. Please close and open again the application (siteurl)');
588                 return false;
589             }
590         }
591         else {
592             var siteurl = preSets.siteurl;
593         }
594 
595         data.wsfunction = method;
596 
597         var ajaxURL = siteurl + '/webservice/rest/server.php?wstoken=' + mytoken + '&moodlewsrestformat=json';
598         var ajaxData = data;
599 
600         // Check if the device is Online, if not add operation to quee.
601         if (preSets.sync) {
602             if (!MM.deviceConnected()) {
603                var el = {
604                 id: hex_md5(ajaxURL + JSON.stringify(ajaxData)),
605                 url: ajaxURL,
606                 data: ajaxData,
607                 syncData: preSets.syncData,
608                 siteid: MM.config.current_site.id,
609                 type: 'ws'
610                };
611                MM.db.insert('sync', el);
612                MM.popMessage(MM.lang.s('addedtoqueue'), {modal: true, title: preSets.syncData.name});
613                return true;
614             }
615         }
616 
617         // Try to get the data from cache.
618         if (preSets.cache) {
619             var omitExpires = false;
620             if (!MM.deviceConnected()) {
621                 // In case the device is not connected, we prefer expired cache than nothing.
622                 omitExpires = true;
623             }
624 
625             var data = MM.cache.getWSCall(ajaxURL, ajaxData, omitExpires);
626 
627             if (data !== false) {
628                 callBack(data);
629                 return true;
630             } else if (!MM.deviceConnected()) {
631                 MM.popErrorMessage(MM.lang.s('networkerrormsg'));
632                 return true;
633             }
634         }
635 
636         // If we arrive here, and we are not connected, thrown a network error message.
637         if (!MM.deviceConnected()) {
638             MM.popErrorMessage(MM.lang.s('networkerrormsg'));
639             return true;
640         }
641 
642         // Main jQuery Ajax call, returns in json format.
643         $.ajax({
644           type: 'POST',
645           url: ajaxURL,
646           data: ajaxData,
647           dataType: 'json',
648           success: function(data) {
649 
650             if (typeof(data.debuginfo) != 'undefined') {
651                 MM.popErrorMessage('Unexpected error. Please close and open again the application');
652                 return;
653             }
654             if (typeof(data.exception) != 'undefined') {
655                 MM.popErrorMessage('Error. ' + data.message);
656                 return;
657             }
658 
659             MM.log('WS: Data received from WS '+ typeof(data));
660 
661             if (typeof(data) == 'object' && typeof(data.length) != 'undefined') {
662                 MM.log('WS: Data number of elements '+ data.length);
663             }
664 
665             if (preSets.cache) {
666                 MM.cache.addWSCall(ajaxURL, ajaxData, data);
667             }
668 
669             // We pass back a clone of the original object, this may prevent erros if in the callback the object is modified.
670             callBack(JSON.parse(JSON.stringify(data)));
671           },
672           error: function(xhr, ajaxOptions, thrownError) {
673             var error = MM.lang.s('cannotconnect');
674             if (xhr.status == 404) {
675                 error = MM.lang.s('invalidscheme');
676             }
677             if (!preSets.silently) {
678                 MM.popErrorMessage(error);
679             } else {
680                 MM.log('WS: error' + method + 'error: '+ error);
681             }
682           }
683         });

769     /**
770      * Launches the WS sync process for operations done offline
771      * There is a queue of tasks performed when the device was offline.
772      */
773     wsSync: function() {
774         MM.log('Sync: Executing WS sync process');
775         if (! MM.getConfig('sync_ws_on')) {
776             MM.log('Sync: WS sync process is disabled');
777         }
778         if (MM.deviceConnected() && MM.getConfig('sync_ws_on')) {
779             MM.db.each('sync', function(sync) {
780                 sync = sync.toJSON();
781                 // Generic call to a WS.
782                 if (sync.type == 'ws') {
783                     MM.log('Sync: Executing WS sync operation:' + JSON.stringify(sync.syncData) + 'url:' + sync.url);
784                     MM.moodleWSCall(sync.data.wsfunction, sync.data, function(d) {
785                         MM.log('Sync: Executing WS sync operation FINISHED:' + sync.data.wsfunction);
786                         MM.db.remove('sync', sync.id);
787                     }, {cache: 0, silently: true});
788                 // File upload.
789                 } else if (sync.type == 'upload') {
790                     MM.log('Sync: Starting upload');
791                     var options = new FileUploadOptions();
792                     options.fileKey = sync.options.fileKey;
793                     options.fileName = sync.options.fileName;
794                     options.mimeType = sync.options.mimeType;
795 
796                     var params = {};
797                     params.token = MM.config.current_token;
798 
799                     options.params = params;
800 
801                     var ft = new FileTransfer();
802                     ft.upload(sync.data, MM.config.current_site.siteurl + '/webservice/upload.php',
803                               function() {
804                                             MM.log('Sync: Execugin Upload sync operation FINISHED:' + sync.options.filename);
805                                             MM.db.remove('sync', sync.id);
806                                         },
807                               function() {
808                                             MM.log('Error uploading');
809                                         },
810                               options);
811                 // Download of contents.
812                 } else if (sync.type == 'content') {
813 
814                     // Only sync files of current site, mainly for performance.
815                     if (sync.siteid == MM.config.current_site.id) {
816 
817                         // Append the token for safe download of files.
818                         sync.url = sync.url + '&token=' + MM.config.current_token;
819 
820                         MM.log('Sync: Starting download of ' + sync.url + ' to ' + sync.newfile);
821                         MM.fs.createDir(sync.path, function() {
822                             MM.moodleDownloadFile(sync.url, sync.newfile,
823                                                   function() {
824                                                     MM.log('Sync: Download of content finished ' + sync.newfile + ' URL: ' + sync.url);
825 
826                                                     var content = MM.db.get('contents', sync.contentid).toJSON();
827                                                     content.contents[sync.index].localpath = sync.newfile;
828                                                     MM.log('Sync: Storing local path in content');
829                                                     MM.db.insert('contents', content);
830 
831                                                     MM.db.remove('sync', sync.id);
832                                                    },
833                                                    function() {
834                                                     MM.log('Sync: Error downloading ' + sync.newfile + ' URL: ' + sync.url);
835                                                     });
836                         });
837                     }
838                 }
839             });
840         }
