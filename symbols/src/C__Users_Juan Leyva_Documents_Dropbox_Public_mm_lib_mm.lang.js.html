 25 /**
 26   * @namespace Holds all the MoodleMobile lang functionallity.
 27  */
 28 MM.lang = {
 29 
 30     strings: [],
 31     current: '',
 32 
 33     determine: function() {
 34         // User preferences.
 35         var lang = MM.getConfig('lang');
 36         if (typeof(lang) != 'undefined') {
 37             return lang;
 38         }
 39 
 40         // Default site lang.
 41         if (typeof(MM.config.current_site) != 'undefined' && typeof(MM.config.current_site.lang) != 'undefined') {
 42             return MM.config.current_site.lang;
 43         }
 44 
 45         // Default language.
 46         return MM.config.default_lang;
 47     },
 48 
 49     setup: function(component) {
 50 
 51         if (typeof(component) == 'undefined') {
 52             component = 'core';
 53         }
 54 
 55         var lang = MM.lang.determine();
 56 
 57         // Try to find in cache the language strings.
 58         // Languages are automatically sync and stored in cache, forcing to not expire.
 59         var langStrings = MM.cache.getElement('lang-' + component + '-' + lang, true);
 60         if (langStrings) {
 61             MM.lang.loadLang(component, lang, langStrings);
 62             MM.log('Strings: Strings loaded from cache (remote syte)');
 63         }
 64     },
 65 
 66     loadLang: function(component, lang, strings) {
 67         MM.log('Strings: Loading lang ' + lang);
 68         MM.lang.current = lang;
 69         if (typeof(MM.lang.strings[lang]) == 'undefined') {
 70             MM.lang.strings[lang] = [];
 71         }
 72         MM.lang.strings[lang][component] = strings;
 73     },
 74 
 75     loadPluginLang: function(component, strings) {
 76         MM.log('Strings: Loading plugin lang ' + component);
 77         if (!MM.lang.current) {
 78             MM.lang.current = 'en';
 79             MM.lang.strings['en'] = [];
 80         }
 81         MM.lang.strings[MM.lang.current][component] = strings;
 82         if (MM.lang.current != 'en') {
 83             MM.lang.strings['en'][component] = strings;
 84         }
 85     },
 86 
 87     pluginName: function(plugin) {
 88 
 89         if (MM.plugins[plugin].settings.lang.component != 'core') {
 90             return MM.lang.s('pluginname', plugin);
 91         }
 92 
 93         return MM.lang.s(plugin);
 94     },
 95 
 96     /**
 97      * Main function for translating strings
 98      *
 99      * @this {MM.lang}
100      * @param {string} id The unique id of the string to be translated.
101      * @param {string} component Core for regular strings or pluginname for plugins.
102      */
103     s: function(id, component) {
104 
105         if (typeof(component) == 'undefined') {
106             component = 'core';
107         }
108 
109         var translated = '';
110         // First we check if we find the string in the current language.
111         if (typeof(MM.lang.strings[MM.lang.current][component]) != 'undefined' && typeof(MM.lang.strings[MM.lang.current][component][id]) !== 'undefined') {
112             translated = MM.lang.strings[MM.lang.current][component][id];
113         }
114         // If not, we look for the string in the default language "english"
115         else if (typeof(MM.lang.strings['en']) != 'undefined'
116                 && typeof(MM.lang.strings['en'][component]) !== 'undefined'
117                 && typeof(MM.lang.strings['en'][component][id]) !== 'undefined') {
118             translated = MM.lang.strings['en'][component][id];
119         }
120 
121         // If not found yet, we look for the string in the base language file (lang/en.json)
122         if (!translated && MM.lang.base[id]) {
123             translated = MM.lang.base[id];
124         }
125 
126         // For missing strings, we use the [string] notation.
127         if (! translated) {
128             translated = '[[' + id + ']]';
129         }
130         return translated;
131     },
132 
133     sync: function() {
134         MM.log('Sync: Executing lang sync function');
135         var lang = MM.lang.determine();
136 
137         if (MM.deviceConnected() && MM.getConfig('sync_lang_on')) {
138             var data = {
139                 'component': 'mobile',
140                 'lang': lang
141             };
142 
143             MM.log('Sync: Loading lang file from remote site for core');
144             MM.moodleWSCall('core_get_component_strings', data, function(strings) {
145                 MM.cache.addElement('lang-core-' + lang, strings, 'lang');
146             }, {silently: true, cache: false});
147 
148             for (var el in MM.plugins) {
149                 var plugin = MM.plugins[el];
150                 var component = plugin.settings.lang.component;
151                 if (component != 'core') {
152                     var data = {
153                         'component': component,
154                         'lang': lang
155                     };
156                     MM.log('Sync: Loading lang from remtote site for component: ' + component);
157                     MM.moodleWSCall('core_get_component_strings', data, function(strings) {
158                         MM.cache.addElement('lang-' + data.component + '-' + lang, strings, 'lang');
159                     }, {silently: true});
160                 }
161             }
162         }
163     }
164 };
