 28 MM.settings = {
 29     display: function() {
 30 
 31         // Settings plugins.
 32         var plugins = [];
 33         for (var el in MM.plugins) {
 34             var plugin = MM.plugins[el];
 35             if (plugin.settings.type == 'setting') {
 36                 plugins.push(plugin.settings);
 37             }
 38         }
 39 
 40         var html = MM.tpl.render($('#settings_template').html(), {plugins: plugins});
 41         MM.panels.show('center', html);
 42         if (MM.deviceType == 'tablet') {
 43             MM.settings.showSection('sites');
 44         }
 45     },
 46 
 47     showSection: function(section) {
 48         // We call dinamically the function.
 49         MM.settings['show' + section.charAt(0).toUpperCase() + section.slice(1)]();
 50     },
 51 
 52     showSites: function() {
 53 
 54         var sites = [];
 55 
 56         MM.db.each('sites', function(el) {
 57             sites.push(el.toJSON());
 58         });
 59 
 60 
 61         MM.collections.sites.fetch();
 62 
 63         var tpl = '\
 64             <div class="sites-list">\
 65             <ul class="nav nav-v">\
 66             <% _.each(sites, function(site){ %>\
 67             <li class="nav-item">\
 68                 <div class="site-info">\
 69                     <a href="#settings/sites/<%= site.id %>"><div class="bd">\
 70                     <h3><%= site.sitename %>\
 71                     <% if (site.id == current_site) { print(" (*)"); }; %>\
 72                     </h3>\
 73                     <%= site.siteurl %> \
 74                     </div></a>\
 75                 </div>\
 76                 <div class="site-actions">\
 77                 <a href="#settings/sites/delete/<%= site.id %>"><img src="img/delete.png" border="0"></a>\
 78                 </div>\
 79             </li>\
 80             <% }); %>\
 81             </ul></div><br \>\
 82             <div class="centered"><a href="#settings/sites/add"><button><%= MM.lang.s("addsite") %></button></a></div>';
 83 
 84         var html = MM.tpl.render(tpl, {sites: sites, current_site: MM.config.current_site.id});
 85         MM.panels.show('right', html);
 86     },
 87 
 88     showSite: function(siteId) {
 89         var site = MM.db.get('sites', siteId);
 90         var options = {
 91             modal: true,
 92             resizable: false,
 93             buttons: {}
 94         };
 95 
 96         var text = '\
 97         <h4>' + site.get('sitename') + '</h4>\
 98         <p><strong>' + MM.lang.s('siteurllabel') + ':</strong> ' + site.get('siteurl') + '</p>\
 99         <p><strong>' + MM.lang.s('fullname') + ':</strong> ' + site.get('fullname') + '</p>\
100         ';
101 
102         if (siteId != MM.config.current_site.id) {
103             options.buttons[MM.lang.s('select')] = function() {
104                 $(this).dialog('close');
105                 MM.loadSite(siteId);
106             };
107         }
108         /*
109         options.buttons[MM.lang.s("delete")] = function() {
110             MM.Router.navigate("settings/sites/delete/" + siteId, {trigger: true, replace: true});
111         };*/
112         options.buttons[MM.lang.s('cancel')] = function() {
113             $(this).dialog('close');
114             MM.Router.navigate('settings/sites/');
115         };
116 
117         MM.widgets.dialog(text, options);
118     },
119 
120     deleteSite: function(siteId) {
121         var site = MM.db.get('sites', siteId);
122         MM.popConfirm(MM.lang.s('deletesite'), function() {
123             var count = MM.db.length('sites');
124             if (count == 1) {
125                 MM.widgets.dialogClose();
126                 MM.popErrorMessage(MM.lang.s('donotdeletesite'));
127             } else {
128                 MM.db.remove('sites', siteId);
129             }
130         });
131     },
132 
133     resetApp: function() {
134         MM.popConfirm(MM.lang.s('areyousurereset'), function() {
135             // Delete all the entries in local storage
136             for (var el in localStorage) {
137                 localStorage.removeItem(el);
138             }
139             // Redirect to maing page
140             location.href = 'index.html';
141         });
142     },
143 
144     showSync: function() {
145         var settings = [
146             {id: 'sync_ws_on', type: 'checkbox', label: MM.lang.s('enableautosyncws'), checked: true, handler: MM.settings.checkboxHandler},
147             {id: 'sync_lang_on', type: 'checkbox', label: MM.lang.s('enableautosynccss'), checked: true, handler: MM.settings.checkboxHandler},
148             {id: 'sync_css_on', type: 'checkbox', label: MM.lang.s('enableautosynclang'), checked: true, handler: MM.settings.checkboxHandler}
149         ];
150 
151         // Load default values
152         $.each(settings, function(index, setting) {
153             if (setting.type == 'checkbox') {
154                 if (typeof(MM.getConfig(setting.id)) != 'undefined') {
155                     settings[index].checked = MM.getConfig(setting.id);
156                 }
157             }
158         });
159 
160         // Render the settings as html.
161         var html = MM.widgets.renderList(settings);
162 
163         var syncFilter = MM.db.where('sync', {siteid: MM.config.current_site.id});
164         var syncTasks = [];
165 
166         $.each(syncFilter, function(index, el) {
167             syncTasks.push(el.toJSON());
168         });
169 
170         var tpl = '\
171             <h2 class="settings-section"><%= MM.lang.s("taskqueue") %></h2>\
172             <% if (tasks.length == 0) { %>\
173             <p><%= MM.lang.s("empty") %></p>\
174             <% } %>\
175             <ul class="nav nav-v">\
176             <% _.each(tasks, function(task){ %>\
177             <li class="nav-item">\
178                 <%= task.syncData.name %><br>\
179                     <i><%= task.syncData.description %></i>\
180             </li>\
181             <% }); %>\
182             </ul><br />\
183             <div class="centered">\
184                 <a href="#settings/sync/lang"><button><%= MM.lang.s("forcelangsync") %></button></a>\
185             </div>\
186             <div class="centered">\
187                 <a href="#settings/sync/css"><button><%= MM.lang.s("forcecsssync") %></button></a>\
188                 <div style="clear: both"></div>\
189             </div>';
190 
191         html += MM.tpl.render(tpl, {tasks: syncTasks});
192 
193         MM.panels.show('right', html);
194         // Once the html is rendered, we pretify the widgets.
195         MM.widgets.enhance(settings);
196         MM.widgets.addHandlers(settings);
197     },

199     showDevelopment: function() {
200 
201         var settingsC = [
202             {id: 'dev_debug', type: 'checkbox', label: MM.lang.s('enabledebugging'), checked: false, handler: MM.settings.checkboxHandler},
203             {id: 'dev_offline', type: 'checkbox', label: MM.lang.s('forceofflinemode'), checked: false, handler: MM.settings.checkboxHandler}
204         ];
205 
206         // Load default values
207         $.each(settingsC, function(index, setting) {
208             if (setting.type == 'checkbox') {
209                 if (typeof(MM.getConfig(setting.id)) != 'undefined') {
210                     settingsC[index].checked = MM.getConfig(setting.id);
211                 }
212             }
213         });
214 
215         var html = MM.widgets.renderList(settingsC);
216 
217         var settingsB = [
218             {id: 'dev_fakenotifications', type: 'button', label: MM.lang.s('addfakenotifications'), handler: MM.settings.addFakeNotifications},
219             {id: 'dev_loadcordovaemulator', type: 'button', label: MM.lang.s('loadcordovaemulator'), handler: MM.cordova.loadEmulator},
220             {id: 'dev_resetapp', type: 'button', label: MM.lang.s('resetapp'), handler: MM.settings.resetApp}
221         ];
222 
223         // Render the settings as html.
224         html += MM.widgets.render(settingsB);
225 
226         // Add the Database manager.
227 
228 
229         MM.panels.show('right', html);
230         // Once the html is rendered, we pretify the widgets.
231         MM.widgets.enhance(settingsC);
232         MM.widgets.addHandlers(settingsC);
233         MM.widgets.enhance(settingsB);
234         MM.widgets.addHandlers(settingsB);
235     },

237     showGeneral: function() {
238 
239         var settings = [
240             {
241                 id: 'cache_expiration_time',
242                 type: 'slider',
243                 config: {
244                     range: 'max',
245                     min: 0,
246                     max: 600,
247                     value: MM.config.cache_expiration_time / 1000,
248                     slide: function(event, ui ) {
249                         $('#cache_expiration_time-text').val(ui.value);
250                         MM.setConfig('cache_expiration_time', parseInt(ui.value + '') * 1000);
251                     }
252                 },
253                 label: MM.lang.s('cacheexpirationtime')
254             }
255         ];
256 
257         // Render the settings as html.
258         var html = MM.widgets.render(settings);
259         html += '<br /><div class="centered"><a href="#settings/general/purgecaches"><button> ' + MM.lang.s('purgecaches') + '</button></a></div>';
260 
261         MM.panels.show('right', html);
262         // Once the html is rendered, we pretify the widgets.
263         MM.widgets.enhance(settings);
264         MM.widgets.addHandlers(settings);
265     },
266 
267     checkboxHandler: function(e, setting) {
268 
269         var val = false;
270         if ($('#' + setting.id).is(':checked')) {
271             val = true;
272         }
273 
274         MM.setConfig(setting.id, val);
275     },

277     addFakeNotifications: function(e, setting) {
278         for (var i = 0; i < 5; i++) {
279             MM.db.insert('notifications', {
280                 siteid: MM.config.current_site.id,
281                 subject: 'Notification ' + i,
282                 date: '23 Dec 2012',
283                 fullmessage: 'Full message of the notification, here we have the full text for this notification. This notification is random created'
284             });
285         }
286         // This is for preserving the Backbone hash navigation.
287         e.preventDefault();
288     },

290     addSite: function(e, setting) {
291 
292         var html = '';
293 
294         var options = {
295             title: MM.lang.s('addsite'),
296             modal: true,
297             buttons: {}
298         };
299         options.buttons[MM.lang.s('add')] = function() {
300             var siteurl = $.trim($('#new-url').val());
301             var username = $.trim($('#new-username').val());
302             var password = $.trim($('#new-password').val());
303 
304             $('form span.error').css('display', 'block').html('');
305 
306             // Delete the last / if present.
307             if (siteurl.charAt(siteurl.length - 1) == '/') {
308                 siteurl = siteurl.substring(0, siteurl.length - 1);
309             }
310 
311             var stop = false;
312 
313             if (siteurl.indexOf('http://localhost') == -1 && ! /^(https?|ftp):\/\/(((([a-z]|\d|-|\.|_|~|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])|(%[\da-f]{2})|[!\$&'\(\)\*\+,;=]|:)*@)?(((\d|[1-9]\d|1\d\d|2[0-4]\d|25[0-5])\.(\d|[1-9]\d|1\d\d|2[0-4]\d|25[0-5])\.(\d|[1-9]\d|1\d\d|2[0-4]\d|25[0-5])\.(\d|[1-9]\d|1\d\d|2[0-4]\d|25[0-5]))|((([a-z]|\d|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])|(([a-z]|\d|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])([a-z]|\d|-|\.|_|~|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])*([a-z]|\d|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])))\.)+(([a-z]|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])|(([a-z]|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])([a-z]|\d|-|\.|_|~|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])*([a-z]|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])))\.?)(:\d*)?)(\/((([a-z]|\d|-|\.|_|~|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])|(%[\da-f]{2})|[!\$&'\(\)\*\+,;=]|:|@)+(\/(([a-z]|\d|-|\.|_|~|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])|(%[\da-f]{2})|[!\$&'\(\)\*\+,;=]|:|@)*)*)?)?(\?((([a-z]|\d|-|\.|_|~|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])|(%[\da-f]{2})|[!\$&'\(\)\*\+,;=]|:|@)|[\uE000-\uF8FF]|\/|\?)*)?(\#((([a-z]|\d|-|\.|_|~|[\u00A0-\uD7FF\uF900-\uFDCF\uFDF0-\uFFEF])|(%[\da-f]{2})|[!\$&'\(\)\*\+,;=]|:|@)|\/|\?)*)?$/i.test(siteurl)) {
314                 stop = true;
315                 $('#new-url').next().html(MM.lang.s('siteurlrequired'));
316             }
317 
318             if (MM.db.get('sites', hex_md5(siteurl + username))) {
319                 stop = true;
320                 $('#new-url').next().html(MM.lang.s('siteexists'));
321             }
322 
323             if (!username) {
324                 stop = true;
325                 $('#new-username').next().html(MM.lang.s('usernamerequired'));
326             }
327             if (!password) {
328                 stop = true;
329                 $('#new-password').next().html(MM.lang.s('passwordrequired'));
330             }
331 
332             if (!stop) {
333                 $(this).dialog('close');
334                 MM.saveSite(username, password, siteurl);
335             }
336         };
337         options.buttons[MM.lang.s('cancel')] = function() {
338             $(this).dialog('close');
339         };
340 
341         html = '\
342         <form>\
343             <p>\
344                 <input type="url" id="new-url" placeholder="' + MM.lang.s('siteurl') + '">\
345                 <span class="error"></div>\
346             </p>\
347             <p>\
348                 <input type="text" id="new-username" placeholder="' + MM.lang.s('username') + '">\
349                 <span class="error"></div>\
350             </p>\
351             <p>\
352                 <input type="password" id="new-password" placeholder="' + MM.lang.s('password') + '">\
353                 <span class="error"></div>\
354             </p>\
355         </form>\
356         ';
357 
358         MM.widgets.dialog(html, options);
359         e.preventDefault();
360     }
