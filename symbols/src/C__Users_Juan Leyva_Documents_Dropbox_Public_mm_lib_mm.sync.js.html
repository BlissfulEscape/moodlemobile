 25 /**
 26   * @namespace Holds all the MoodleMobile sync functionallity.
 27  */
 28 MM.sync = {
 29     interval: 0,
 30     hooks: {},
 31 
 32     init: function() {
 33 
 34         MM.sync.hooks = {
 35             'ws': {
 36                 handler: MM.wsSync,
 37                 time: MM.config.sync_ws
 38             },
 39             'lang': {
 40                 handler: MM.lang.sync,
 41                 time: MM.config.sync_lang
 42             },
 43             'css': {
 44                 handler: MM.sync.css,
 45                 time: MM.config.sync_css
 46             }
 47         };
 48 
 49         // We need to load the FS before starting sync processes.
 50         MM.fs.init(function() {
 51             MM.sync.interval = parseInt(MM.config.sync_cron + '');
 52             MM.log('Sync: process starting in...' + MM.config.sync_cron + ' milliseconds');
 53             // If we get a integer setting as a string, we force to string allways and parses to int
 54             setTimeout(function() { MM.sync.syncProcess()}, MM.sync.interval);
 55         });
 56     },
 57 
 58     syncProcess: function() {
 59         var newInterval = 0;
 60         var nextExecution = 0;
 61         var d = new Date();
 62         var call = false;
 63 
 64         MM.log('Sync: process executing, current time: ' + d.getTime());
 65 
 66         for (var el in MM.sync.hooks) {
 67             call = false;
 68             var lastExecution = MM.db.get('settings', 'last_sync' + el);
 69 
 70             if (!lastExecution) {
 71                 call = true;
 72             } else {
 73                 MM.log('Sync: ' + el + ' last execution: ' + lastExecution.get('value') + ', Time: ' + MM.sync.hooks[el].time);
 74                 nextExecution = parseInt(lastExecution.get('value') + '') + MM.sync.hooks[el].time;
 75                 MM.log('Sync: ' + el + ' current time ' + d.getTime() + ', next execution time :' + nextExecution);
 76                 if (d.getTime() > nextExecution) {
 77                     call = true;
 78                 }
 79             }
 80 
 81             if (call) {
 82                 MM.sync.callHandler(el, MM.sync.hooks[el]);
 83                 newInterval = (newInterval && newInterval > MM.sync.hooks[el].time) ? MM.sync.hooks[el].time : newInterval;
 84             } else if (lastExecution) {
 85                 if (lastExecution.get('value') + newInterval > lastExecution.get('value') + MM.sync.hooks[el].time) {
 86                     newInterval = MM.sync.hooks[el].time;
 87                 }
 88             }
 89         }
 90 
 91         newInterval = (!newInterval) ? MM.sync.interval : newInterval;
 92         MM.log('Sync: new Interval for sync process: ' + newInterval);
 93         setTimeout(function() { MM.sync.syncProcess()}, newInterval);
 94     },
 95 
 96     callHandler: function(el, hook) {
 97         MM.log('Sync: calling hooks for: ' + el);
 98         var d = new Date();
 99         var newTime = d.getTime();
100         MM.setConfig('last_sync' + el, newTime);
101         MM.log('Sync: '+ el + ' setting new execution time: ' + newTime);
102         MM.sync.hooks[el].handler();
103     },
104 
105     /**
106      * Function for register hooks from plugins.
107      */
108     registerHook: function(name, hook) {
109         MM.sync.hooks[name] = hook;
110     },
111 
112     css: function() {
113         MM.log('Executing css sync function');
114         if (MM.deviceConnected()
115                 && typeof(MM.config.current_site.mobilecssurl) != 'undefined'
116                 && MM.config.current_site.mobilecssurl
117                 && MM.getConfig('sync_css_on')) {
118 
119             var cssURL = MM.config.current_site.mobilecssurl;
120             $.ajax({
121                 url: cssURL,
122                 success: function(data) {
123                     MM.cache.addElement('css', data, 'css');
124                     $('#mobilecssurl').html(data);
125                 },
126                 error: function(xhr, ajaxOptions, thrownError) {
127                   var error = MM.lang.s('cannotconnect');
128                   if (xhr.status == 404) {
129                       error = MM.lang.s('invalidscheme');
130                   }
131                   MM.log('Sync; Error downloading CSS' + error);
132                 }
133             });
134         }
135     }
136 };
