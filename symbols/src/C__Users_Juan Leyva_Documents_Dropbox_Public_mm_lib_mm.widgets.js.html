 25 /**
 26   * @namespace Holds all the MoodleMobile html widgets functionallity.
 27  */
 28 MM.widgets = {
 29         
 30     render: function(elements) {
 31         var settings = [];
 32         var output = "";
 33         
 34         if($.isArray(elements)) {
 35             settings = elements;
 36         } else {
 37             settings.push(elements);
 38         }
 39         
 40         $.each(settings, function(index, setting) {
 41             var renderer = "render" + setting.type.charAt(0).toUpperCase() + setting.type.slice(1);
 42             if (typeof MM.widgets[renderer] != "undefined") {
 43                 output += "<div class=\"mm-setting\">" + MM.widgets[renderer](setting) + "</div>";
 44             }
 45         });
 46         return output;
 47     },
 48 
 49     renderList: function(elements) {
 50         var settings = [];
 51         var output = '<ul class="nav nav-v">';
 52         
 53         if($.isArray(elements)) {
 54             settings = elements;
 55         } else {
 56             settings.push(elements);
 57         }
 58         
 59         $.each(settings, function(index, setting) {
 60             var renderer = "render" + setting.type.charAt(0).toUpperCase() + setting.type.slice(1);
 61             if (typeof MM.widgets[renderer] != "undefined") {
 62                 output += '<li class="nav-item">' + MM.widgets[renderer](setting) + '</li>';
 63             }
 64         });
 65         
 66         output += "</ul>";
 67         
 68         return output;
 69     },
 70     
 71     renderCheckbox: function(el) {
 72         if (el.checked) {
 73             el.checked = 'checked = "checked"';
 74         }
 75         var tpl = '<div class="checkbox-setting"><div class="checkbox-label"> <%= label %></div><div class="checkbox-tic"><input type="checkbox" id="<%= id %>" <%= checked %>/>\
 76                    <label for="<%= id %>"></label></div></div>';
 77         return MM.tpl.render(tpl, el);
 78     },
 79 
 80     renderButton: function(el) {
 81         if (typeof el.url == "undefined") {
 82             el.url = "";
 83         }
 84         var tpl = '<a href="#<%= url %>" id="<%= id %>"><button><%= label %></button></a>';
 85         return MM.tpl.render(tpl, el);
 86     },
 87     
 88     renderSlider: function(el) {
 89         var tpl = '\
 90         <p><label for="<%= id %>-text"><%= label %></label>\
 91         <input type="text" id="<%= id %>-text" style="border: 0; color: #f6931f; font-weight: bold;" />\
 92         </p><div id="<%= id %>"></div>';
 93         return MM.tpl.render(tpl, el);
 94     },
 95 
 96     enhanceButton: function(id) {
 97         //$("#" + id).button();
 98     },
 99 
100     enhanceCheckbox: function(id) {
101         //$("#" + id).parent().addClass("checkbox-tic");
102     },
103     
104     enhanceSlider: function(id, config) {
105         $( "#" + id ).slider(config);
106         $( "#" + id + "-text" ).val( config.value );        
107     },
108     
109     enhance: function(elements) {
110         var settings = [];
111         var output = "";
112         
113         if($.isArray(elements)) {
114             settings = elements;
115         } else {
116             settings.push(elements);
117         }
118         
119         $.each(settings, function(index, setting) {
120             var enhancer = "enhance" + setting.type.charAt(0).toUpperCase() + setting.type.slice(1);
121             if (typeof MM.widgets[enhancer] != "undefined") {
122                 if (setting.config) {
123                     MM.widgets[enhancer](setting.id, setting.config);
124                 } else {
125                     MM.widgets[enhancer](setting.id);
126                 }
127             }
128         });
129         //MM.widgets.improveCheckbox();
130     },
131 
132     addHandlers: function(elements) {
133         var settings = [];
134         var output = "";
135 
136         if($.isArray(elements)) {
137             settings = elements;
138         } else {
139             settings.push(elements);
140         }
141         
142         $.each(settings, function(index, setting) {
143             if (typeof setting.handler != "undefined") {
144                 var fn = setting.handler;
145                 $("#" + setting.id).click(function(e){
146                     fn(e, setting);
147                 });
148             }
149         });
150     },
151     
152     dialog: function(text, options) {
153         var dialog = $("#app-dialog");
154         dialog.html(text).dialog(options);
155         if (options && options.autoclose) {
156             setTimeout(function() { dialog.dialog("close");}, options.autoclose);
157         }
158     },
159     
160     dialogClose: function() {
161         var dialog = $("#app-dialog");
162         dialog.dialog("close");
163         dialog.dialog("destroy");
164     },
165     
166     improveCheckbox: function() {
167         var onClass = "ui-icon-circle-check", offClass = "ui-icon-circle-close";
168         
169         $( "input:checked[type=checkbox] " ).button({ icons: {primary: onClass} });
170         $( "input[type=checkbox]:not(:checked)" ).button({ icons: {primary: offClass} });
171         
172         $( "input[type=checkbox]" ).click(function(){
173 
174             var swap = function(me, toAdd, toRemove) {
175                 // find the LABEL for the checkbox
176                 // ... which should be _immediately_ before or after the checkbox
177                 var node = me.next();
178 
179                 // and swap
180                 node.children(".ui-button-icon-primary")
181                     .addClass(toAdd)
182                     .removeClass(toRemove);
183                 ;
184             };
185         
186             var me = $(this);
187             if (me.is(':checked')) {
188                 swap($(this), onClass, offClass);
189             } else {
190                 swap($(this), offClass, onClass);
191             }
192         });        
193     }
194 }
