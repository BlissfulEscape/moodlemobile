 25 function isTouchDevice() {
 26 	try {
 27 		document.createEvent('TouchEvent');
 28 		return true;
 29 	}catch (e) {
 30 		return false;
 31 	}
 32 }
 33 
 34 function touchScroll(id) {
 35 
 36 	if (isTouchDevice()) { //if touch events exist...
 37 		var el = document.getElementById(id);
 38 		var scrollStartPos = 0;
 39 
 40 		document.getElementById(id).addEventListener('touchstart', function(event) {
 41 			scrollStartPos = this.scrollTop + event.touches[0].pageY;
 42 		},false);
 43 
 44 		document.getElementById(id).addEventListener('touchmove', function(event) {
 45 			this.scrollTop = scrollStartPos - event.touches[0].pageY;
 46 		},false);
 47 	}
 48 }
